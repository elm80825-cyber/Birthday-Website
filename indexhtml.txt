<script>
// Three.js Background
const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
const renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
document.getElementById('canvas-container').appendChild(renderer.domElement);

const hearts = [];
const geometry = new THREE.IcosahedronGeometry(0.5, 2);
const material = new THREE.MeshPhongMaterial({ color: 0xff69b4, shininess: 100 });

for(let i=0; i<25; i++) {
const heart = new THREE.Mesh(geometry, material);
heart.position.set((Math.random()-0.5)*10, (Math.random()-0.5)*10, -5);
scene.add(heart);
hearts.push(heart);
}
const light = new THREE.PointLight(0xffffff, 1, 100);
light.position.set(0, 0, 5);
scene.add(light);
camera.position.z = 5;

function animate() {
requestAnimationFrame(animate);
hearts.forEach((h, i) => { h.rotation.y += 0.01; h.position.y += Math.sin(Date.now()*0.001 + i) * 0.01; });
renderer.render(scene, camera);
}
animate();

// Step Logic
let currentStep = 1;
const steps = [
{ id: 1, icon: "❤️", title: "Hey Beautiful,", body: "I built a little world for you, just to bring a smile to your face on your special day.", btn: "Let's Begin" },
{ id: 2, icon: "🎉", title: "Happy Birthday!", body: "Another year of you making the world brighter. Your existence is a gift, and I'm so lucky to witness it.", btn: "There's more..." },
{ id: 3, type: 'bento', title: "A Few Things I Adore About You", items: [
{t: "✨ Your Kindness", d: "The genuine warmth you show to everyone is truly rare."},
{t: "😊 That Smile", d: "It's a work of art."},
{t: "🌟 Your Spirit", d: "Your passion for life is infectious."}
 ], btn: "Remember this?" },
{ id: 4, type: 'polaroid', title: "That One Time...", img: "https://i.ibb.co/6Z6XgCg/crush.webp", body: "Every moment with you feels like a scene from a movie I'd watch on repeat.", btn: "One last thing..." },
{ id: 5, icon: "🎂", title: "My Wish For You", body: "May the next year bring you all the love, success, and pure happiness you so rightfully deserve.", btn: "Celebrate!" }
];

function renderStep(id) {
const step = steps.find(s => s.id === id);
const app = document.getElementById('app');
document.getElementById('progress').style.width = ${(id/5)*100}%`;

let html = <div id="step-content" class="glass p-8 md:p-12 max-w-lg w-full text-center transition-all duration-700">;

if(step.icon) html += <div class="text-6xl mb-6 animate-bounce"&gt;${step.icon}</div>; if(step.type === 'bento') { html +=<h2 class="text-3xl text-gradient mb-8">{step.items.map((it,i) => <div class="glass p-4 ${i===0?'col-span-2':''}"><h3 class="font-bold">${it.t}&lt;/h3&gt;&lt;p class="text-sm opacity-80"&gt;${it.d}</p></div>).join('')} &lt;/div>;
} else if(step.type === 'polaroid') {
html += <h2 class="text-3xl text-gradient mb-4"&gt;${step.title}</h2>
<div class="bg-white p-4 pb-12 rotate-2 mb-6 inline-block shadow-xl"><img src="{step.body}</p>; } else { html +=<h2 class="text-4xl text-gradient mb-4">{step.body}</p>`;
}

html += <button onclick="nextStep()" class="btn-grad px-8 py-3 rounded-full font-bold text-lg w-full"&gt;${step.btn}&lt;/button&gt;&lt;/div>;
app.innerHTML = html;
gsap.fromTo("#step-content", {y: 50, opacity: 0}, {y: 0, opacity: 1, duration: 1});
}

function nextStep() {
if(currentStep < 5) {
currentStep++;
renderStep(currentStep);
} else {
// Finale
document.getElementById('app').innerHTML = <h1 class="text-5xl text-gradient font-bold animate-pulse"&gt;Happy Birthday, my crush! ❤️&lt;/h1>;
confetti({ particleCount: 200, spread: 100, origin: { y: 0.6 } });
// Swirl hearts away
gsap.to(hearts.map(h => h.position), { y: 20, duration: 3, stagger: 0.05 });
}
}

renderStep(1);
</script>
